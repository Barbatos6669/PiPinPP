356 unsigned long pulseIn(int pin, bool state, unsigned long timeout)
357 {
358  // Validate pin number first (before acquiring mutex)
359  if (pin < 0 || pin > 27) {
360  throw InvalidPinError("Invalid pin number: " + std::to_string(pin) +
361  ". Valid range is 0-27.");
362  }
363 
364  // Verify pin is INPUT (don't auto-configure to avoid deadlock)
365  {
366  std::lock_guard<std::mutex> lock(globalPinsMutex);
367  auto it = globalPins.find(pin);
368  if (it == globalPins.end()) {
369  throw PinError("Pin " + std::to_string(pin) +
370  " not initialized. Call pinMode(pin, INPUT) first.");
371  } else if (it->second.mode == ArduinoPinMode::OUTPUT) {
372  throw PinError("Pin " + std::to_string(pin) + " is configured as OUTPUT. "
373  "Cannot use pulseIn() on output pins.");
374  }
375  }
376 
377  auto start = std::chrono::steady_clock::now();
378  auto timeoutDuration = std::chrono::microseconds(timeout);
379 
380  // Wait for pin to go to opposite state (start of pulse)
381  while (digitalRead(pin) == state) {
382  if (std::chrono::steady_clock::now() - start >= timeoutDuration) {
383  return 0; // Timeout waiting for pulse start
384  }
385  }
386 
387  // Wait for pin to change to desired state (beginning of pulse)
388  while (digitalRead(pin) != state) {
389  if (std::chrono::steady_clock::now() - start >= timeoutDuration) {
390  return 0; // Timeout waiting for pulse
391  }
392  }
393 
394  // Measure pulse duration
395  auto pulseStart = std::chrono::steady_clock::now();
396  while (digitalRead(pin) == state) {
397  if (std::chrono::steady_clock::now() - start >= timeoutDuration) {
398  return 0; // Timeout during pulse
399  }
400  }
401  auto pulseEnd = std::chrono::steady_clock::now();
402 
403  auto duration = std::chrono::duration_cast<std::chrono::microseconds>(pulseEnd - pulseStart);
404  return static_cast<unsigned long>(duration.count());
405 }

408 {
409  // Validate pin numbers first (before acquiring mutex)
410  if (dataPin < 0 || dataPin > 27) {
411  throw InvalidPinError("Invalid data pin number: " + std::to_string(dataPin) +
412  ". Valid range is 0-27.");
413  }
414  if (clockPin < 0 || clockPin > 27) {
415  throw InvalidPinError("Invalid clock pin number: " + std::to_string(clockPin) +
416  ". Valid range is 0-27.");
417  }
418 
419  // Verify pins are OUTPUT. Don't auto-configure with pinMode() here, as that
420  // would acquire globalPinsMutex while it is already held, causing a deadlock.
421  // (pinMode() locks globalPinsMutex internally.)
422  {
423  std::lock_guard<std::mutex> lock(globalPinsMutex);
424  if (globalPins.find(dataPin) == globalPins.end() ||
425  globalPins[dataPin].mode != ArduinoPinMode::OUTPUT) {
426  throw PinError("Pin " + std::to_string(dataPin) +
427  " must be OUTPUT for shiftOut(). Call pinMode() first.");
428  }

430  globalPins[clockPin].mode != ArduinoPinMode::OUTPUT) {
431  throw PinError("Pin " + std::to_string(clockPin) +
432  " must be OUTPUT for shiftOut(). Call pinMode() first.");
433  }
434  }
435 
436  for (int i = 0; i < 8; i++) {
437  bool bitValue;
438  if (bitOrder == LSBFIRST) {
439  bitValue = !!(value & (1 << i)); // LSB first
440  } else {
441  bitValue = !!(value & (1 << (7 - i))); // MSB first
442  }
443 
444  digitalWrite(dataPin, bitValue);
445  digitalWrite(clockPin, HIGH);

448  digitalWrite(clockPin, LOW);
