210 // Detect GPIO chips
211 void PlatformInfo::detectGPIOChips()
212 {
213  capabilities_.gpioChips.clear();
214 
215  // Enumerate all gpiochip devices
216  for (int i = 0; i < 16; i++) { // Check up to gpiochip15
217  std::string chipName = "gpiochip" + std::to_string(i);
218  std::string chipPath = "/dev/" + chipName;
219 
220  if (!deviceExists(chipPath)) {
221  continue;
222  }
223 
224  // Open chip to get info (libgpiod v2 API)
225  struct gpiod_chip* chip = gpiod_chip_open(chipPath.c_str());
226  if (!chip) {
227  continue;
228  }
229 
230  // Get chip info
231  struct gpiod_chip_info* chip_info = gpiod_chip_get_info(chip);
232  if (!chip_info) {
233  gpiod_chip_close(chip);
234  continue;
235  }
236 
237  GPIOChipInfo info;
238  info.name = chipName;
239  info.label = gpiod_chip_info_get_label(chip_info);
240  info.numLines = gpiod_chip_info_get_num_lines(chip_info);
241  info.available = true;
242 
243  capabilities_.gpioChips.push_back(info);
244 
245  gpiod_chip_info_free(chip_info);
246  gpiod_chip_close(chip);
247  }
248 }

270 // Detect PWM channels
271 void PlatformInfo::detectPWMChannels()
272 {
273  capabilities_.pwmChannels.clear();
274 
275  // Check sysfs PWM chips
276  DIR* dir = opendir("/sys/class/pwm");
277  if (!dir) {
278  return; // No PWM support or permission denied
279  }
280 
281  struct dirent* entry;
282  while ((entry = readdir(dir)) != nullptr) {
283  std::string name = entry->d_name;
284 
285  // Look for pwmchip* entries
286  if (name.find("pwmchip") == 0) {
287  int chipNum = std::stoi(name.substr(7)); // Extract number
288 
289  std::string sysfsPath = "/sys/class/pwm/" + name;
290  std::string npwmPath = sysfsPath + "/npwm";
291 
292  // Read number of PWM channels
293  std::string npwmStr = readFile(npwmPath);
294  if (npwmStr.empty()) {
295  continue;
296  }
297 
298  int numChannels = std::stoi(npwmStr);
299 
300  // Create info for each channel
301  for (int ch = 0; ch < numChannels; ch++) {
302  PWMChannelInfo info;
303  info.chip = chipNum;
304  info.channel = ch;
305  info.gpioPin = -1; // TODO: Map to GPIO pin number
306  info.sysfsPath = sysfsPath;
307  info.available = (access(sysfsPath.c_str(), R_OK | W_OK) == 0);
308 
309  capabilities_.pwmChannels.push_back(info);
310  }
311  }
312  }
313 
314  closedir(dir);
315 
316  // Map PWM channels to GPIO pins (Raspberry Pi specific)
317  if (isRaspberryPi()) {
318  // Pi 4/5: PWM0 = GPIO18, PWM1 = GPIO19 (common mapping)
319  for (auto& pwm : capabilities_.pwmChannels) {
320  if (pwm.chip == 0 && pwm.channel == 0) {
321  pwm.gpioPin = 18;
322  } else if (pwm.chip == 0 && pwm.channel == 1) {
323  pwm.gpioPin = 19;
324  }
325  }
326  }
327 }

347 void PlatformInfo::detectDMASupport()
348 {
349  capabilities_.hasDMASupport = false;
350  capabilities_.peripheralBase = 0;
351 
352  // Only Raspberry Pi supports DMA GPIO (for now)
353  if (!isRaspberryPi()) {
354  return;
355  }
356 
357  // Check if /dev/mem exists and is readable
358  if (!fileExists("/dev/mem")) {
359  return;
360  }
361

367  return;

400 // Parse /proc/cpuinfo
401 Platform PlatformInfo::parseCPUInfo() const
402 {
403  std::string cpuinfo = readFile("/proc/cpuinfo");
404  if (cpuinfo.empty()) {
405  return Platform::UNKNOWN;
406  }
407 
408  // Look for Raspberry Pi indicators
409  if (cpuinfo.find("BCM2835") != std::string::npos) {
410  return Platform::RASPBERRY_PI_ZERO;
411  }
412  if (cpuinfo.find("BCM2836") != std::string::npos) {
413  return Platform::RASPBERRY_PI_3; // Pi 2 v1.2+ shares BCM2836
414  }
415  if (cpuinfo.find("BCM2837") != std::string::npos) {
416  // Could be Pi 3 or Zero 2
417  if (cpuinfo.find("Raspberry Pi Zero 2") != std::string::npos) {
418  return Platform::RASPBERRY_PI_ZERO2;
419  }
420  return Platform::RASPBERRY_PI_3;
421  }
422  if (cpuinfo.find("BCM2711") != std::string::npos) {
423  // Could be Pi 4 or CM4
424  if (cpuinfo.find("Compute Module 4") != std::string::npos) {
425  return Platform::RASPBERRY_PI_CM4;
426  }
427  return Platform::RASPBERRY_PI_4;
428  }
429  if (cpuinfo.find("BCM2712") != std::string::npos) {
430  return Platform::RASPBERRY_PI_5;
431  }
432 
433  // Orange Pi indicators
434  if (cpuinfo.find("Allwinner") != std::string::npos ||
435  cpuinfo.find("sun") != std::string::npos) { // Allwinner SoCs often have "sun" prefix
436  return Platform::ORANGE_PI;
437  }
438 
439  // BeagleBone indicators
440  if (cpuinfo.find("AM33XX") != std::string::npos ||
441  cpuinfo.find("AM335") != std::string::npos) {
442  return Platform::BEAGLEBONE;
443  }
444 
445  // Jetson Nano indicators
446  if (cpuinfo.find("Tegra") != std::string::npos) {
447  return Platform::JETSON_NANO;
448  }
449 
450  return Platform::UNKNOWN;
451 }

504 // Get peripheral base address
505 uint32_t PlatformInfo::getPeripheralBase() const
506 {
507  switch (platform_) {
508  case Platform::RASPBERRY_PI_ZERO:
509  case Platform::RASPBERRY_PI_3:
510  case Platform::RASPBERRY_PI_ZERO2:
511  return 0x3F000000; // BCM2835/2837
512 
513  case Platform::RASPBERRY_PI_4:
514  case Platform::RASPBERRY_PI_CM4:
515  return 0xFE000000; // BCM2711
516

518  // Pi 5 uses 64-bit addressing with RP1 I/O controller

520  return 0x1F00000000ULL & 0xFFFFFFFF; // Truncated for 32-bit compatibility
521 
522  default:
523  return 0; // Unknown platform
524  }
