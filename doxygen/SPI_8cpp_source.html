215 bool SPIClass::applySettings() {
216  // Note: Mutex should already be locked by caller
217 
218  if (fd_ < 0) {
219  return false;
220  }
221 
222  // Set SPI mode
223  uint8_t spiMode = mode_;
224  if (bitOrder_ == LSBFIRST) {
225  spiMode |= SPI_LSB_FIRST;
226  }
227 
228  if (ioctl(fd_, SPI_IOC_WR_MODE, &spiMode) < 0) {
229  return false;
230  }
231 
232  // Set bits per word
233  if (ioctl(fd_, SPI_IOC_WR_BITS_PER_WORD, &bitsPerWord_) < 0) {
234  return false;
235  }
236 
237  // Set max speed

239  return false;
240  }
