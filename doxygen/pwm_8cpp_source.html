168 void PWMManager::pwmThreadFunction(PWMChannel* channel) {
169  PIPINPP_LOG_DEBUG("PWM thread started for pin " << channel->pin);
170 
171  while (channel->active) {
172  int duty = channel->dutyCycle;
173  int freq = channel->frequency;
174 
175  // Calculate period and timing
176  double periodUs = 1000000.0 / freq; // Period in microseconds
177  double onTimeUs = (periodUs * duty) / 255.0;
178 
179  // Handle edge cases
180  if (duty == 0) {
181  // Always off
182  channel->pinObj->write(false);
183  std::this_thread::sleep_for(std::chrono::milliseconds(10));
184  continue;
185  } else if (duty == 255) {
186  // Always on
187  channel->pinObj->write(true);
188  std::this_thread::sleep_for(std::chrono::milliseconds(10));
189  continue;
190  }
191 
192  // Generate PWM pulse
193  auto cycleStart = std::chrono::high_resolution_clock::now();
194 
195  // HIGH period
196  channel->pinObj->write(true);
197  auto onTarget = cycleStart + std::chrono::nanoseconds(static_cast<long>(onTimeUs * 1000));
198  while (std::chrono::high_resolution_clock::now() < onTarget && channel->active) {
199  // Busy-wait for precision (necessary for software PWM)
200  std::this_thread::yield();
201  }
202 
203  // LOW period
204  channel->pinObj->write(false);
205  auto offTarget = cycleStart + std::chrono::nanoseconds(static_cast<long>(periodUs * 1000));
206  while (std::chrono::high_resolution_clock::now() < offTarget && channel->active) {
207  // Busy-wait for precision
208  std::this_thread::yield();
209  }
210  }
211 
212  // Ensure pin is LOW when stopping
213  if (channel->pinObj) {
214  channel->pinObj->write(false);
215  }
216 
217  PIPINPP_LOG_DEBUG("PWM thread stopped for pin " << channel->pin);
218 }
