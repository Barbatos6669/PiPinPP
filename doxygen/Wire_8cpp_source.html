139 uint8_t WireClass::endTransmission(bool sendStop) {
140  (void)sendStop; // Parameter not used on Linux i2c-dev (always sends STOP)
141  std::lock_guard<std::mutex> lock(mutex_);
142 
143  if (fd_ < 0) {
144  return 4; // Other error (not initialized)
145  }
146 
147  if (txBuffer_.empty()) {
148  return 0; // Success (nothing to send)
149  }
150 
151  if (!setSlaveAddress(txAddress_)) {
152  return 2; // NACK on address
153  }
154 
155  // Write data
156  ssize_t written = ::write(fd_, txBuffer_.data(), txBuffer_.size());
157 
158  if (written < 0) {
159  return 3; // NACK on data
160  }
161 
162  if (static_cast<size_t>(written) != txBuffer_.size()) {
163  return 3; // Incomplete write
164  }
165 
166  txBuffer_.clear();
167  return 0; // Success
168 }

189 size_t WireClass::write(const uint8_t* data, size_t length) {
190  std::lock_guard<std::mutex> lock(mutex_);
191 
192  size_t available = BUFFER_SIZE - txBuffer_.size();
193  size_t toWrite = (length < available) ? length : available;
194 
195  if (toWrite == 0) {

207 size_t WireClass::requestFrom(uint8_t address, size_t quantity, bool sendStop) {
208  (void)sendStop; // Parameter not used on Linux i2c-dev (always sends STOP)
209  std::lock_guard<std::mutex> lock(mutex_);
210 
211  if (fd_ < 0) {
212  return 0;
213  }
214 
215  if (quantity == 0 || quantity > BUFFER_SIZE) {
216  return 0;
217  }
218 
219  if (!setSlaveAddress(address)) {
220  return 0;
221  }
222 
223  // Prepare receive buffer
224  rxBuffer_.resize(quantity);
225  rxIndex_ = 0;
226 
227  // Read data
228  ssize_t bytesRead = ::read(fd_, rxBuffer_.data(), quantity);
229 
230  if (bytesRead < 0) {
231  rxBuffer_.clear();
232  return 0;
233  }
234 
235  rxBuffer_.resize(bytesRead);
236  return bytesRead;
237 }

299 int WireClass::readRegister(uint8_t address, uint8_t reg) {

311  return read();

322 int WireClass::readRegisters(uint8_t address, uint8_t reg, uint8_t* buffer, size_t length) {
323  if (buffer == nullptr || length == 0) {
324  return -1;
325  }
326 
327  beginTransmission(address);
328  write(reg);
329 
330  if (this->endTransmission(false) != 0) { // Repeated START
331  return -1;
332  }
333 
334  size_t received = this->requestFrom(address, length, true);
335 
336  for (size_t i = 0; i < received; i++) {
337  int byte = read();
338  if (byte < 0) {
339  return i; // Return how many we got
340  }
341  buffer[i] = static_cast<uint8_t>(byte);
342  }
343 
344  return received;
345 }

371 int WireClass::detectBusNumber() {
372  // Try common I2C bus numbers
373  // Pi 0-4: typically /dev/i2c-1
374  // Pi 5: typically /dev/i2c-20 or /dev/i2c-21
375 
376  const int candidates[] = {1, 20, 21, 0, 2};
377 
378  for (int bus : candidates) {
379  char filename[20];
380  snprintf(filename, sizeof(filename), "/dev/i2c-%d", bus);
381 
382  struct stat st;
383  if (stat(filename, &st) == 0) {
384  return bus;
385  }
386  }
387 
388  return -1; // No I2C device found
389 }
