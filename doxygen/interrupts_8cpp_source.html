268 void InterruptManager::monitorThread() {
269  PIPINPP_LOG_DEBUG("Monitor thread running");
270 
271  while (!shutdown_requested_) {
272  std::vector<pollfd> poll_fds;
273  std::vector<int> pin_map; // Maps poll_fds index to pin number
274 
275  {
276  std::lock_guard<std::mutex> lock(mutex_);
277 
278  // Add wakeup pipe
279  pollfd wakeup_fd;
280  wakeup_fd.fd = wakeup_pipe_[0];
281  wakeup_fd.events = POLLIN;
282  wakeup_fd.revents = 0;
283  poll_fds.push_back(wakeup_fd);
284  pin_map.push_back(-1); // -1 indicates wakeup pipe
285 
286  // Add all active interrupt handlers
287  for (auto& pair : handlers_) {
288  if (pair.second->active) {
289  pollfd pfd;
290  pfd.fd = gpiod_line_request_get_fd(pair.second->request);
291  pfd.events = POLLIN;
292  pfd.revents = 0;
293  poll_fds.push_back(pfd);
294  pin_map.push_back(pair.first);
295  }
296  }
297  }
298 
299  if (poll_fds.size() == 1) {
300  // Only wakeup pipe, no active interrupts
301  // Wait for new interrupts to be attached
302  poll(poll_fds.data(), poll_fds.size(), -1);
303 
304  // Clear wakeup pipe
305  char buf[16];
306  ssize_t bytes_read = read(wakeup_pipe_[0], buf, sizeof(buf));
307  (void)bytes_read; // Intentionally ignore - just clearing the pipe
308  continue;
309  }
310 
311  // Wait for events (timeout 100ms to check shutdown flag periodically)
312  int ret = poll(poll_fds.data(), poll_fds.size(), 100);
313 
314  if (ret < 0) {
315  if (errno == EINTR) {
316  continue; // Interrupted by signal, retry
317  }
318  PIPINPP_LOG_ERROR("poll() failed: " << strerror(errno));
319  break;
320  }
321 
322  if (ret == 0) {
323  continue; // Timeout, check shutdown flag
324  }
325 
326  // Check wakeup pipe first
327  if (poll_fds[0].revents & POLLIN) {
328  char buf[16];
329  ssize_t bytes_read = read(wakeup_pipe_[0], buf, sizeof(buf));
330  (void)bytes_read; // Intentionally ignore - just clearing the pipe
331  // Reconfigure poll on next iteration
332  continue;
333  }
334 
335  // Process events on interrupt pins
336  for (size_t i = 1; i < poll_fds.size(); ++i) {
337  if (poll_fds[i].revents & POLLIN) {
338  int pin = pin_map[i];
339 
340  std::unique_lock<std::mutex> lock(mutex_);
341  auto it = handlers_.find(pin);
342  if (it == handlers_.end() || !it->second->active) {
343  continue; // Handler was removed
344  }
345 
346  auto& handler = it->second;
347  lock.unlock();
348 
349  // Read edge events
350  int num_events = gpiod_line_request_read_edge_events(
351  handler->request, handler->event_buffer,
352  gpiod_edge_event_buffer_get_capacity(handler->event_buffer)
353  );
354 
355  if (num_events > 0) {
356  // Process all events
357  for (int j = 0; j < num_events; ++j) {
358  gpiod_edge_event* event = gpiod_edge_event_buffer_get_event(
359  handler->event_buffer, j
360  );
361 
362  if (event) {
363  // Invoke callback
364  try {
365  handler->callback();
366  } catch (const std::exception& e) {
367  PIPINPP_LOG_ERROR("Exception in interrupt callback for pin "
368  << pin << ": " << e.what());
369  } catch (...) {

371  }
372  }
373  }
374  }
375  }
376  }
377  }
