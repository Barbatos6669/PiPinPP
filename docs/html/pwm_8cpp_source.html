55void PWMManager::startPWM(int pin, int dutyCycle, int frequency) {
56 // Validate pin number
57 if (pin < 0 || pin > 27) {
58 throw InvalidPinError("Invalid pin number: " + std::to_string(pin) +
59 " (must be 0-27 for Raspberry Pi)");
60 }
61
62 // Validate and clamp values
63 dutyCycle = validateDutyCycle(dutyCycle);
64 frequency = validateFrequency(frequency);
65
66 std::lock_guard<std::mutex> lock(mutex_);
67
68 // Check if PWM already exists on this pin
69 auto it = channels_.find(pin);
70 if (it != channels_.end()) {
71 // Update duty cycle only
72 it->second->dutyCycle = dutyCycle;
73 PIPINPP_LOG_INFO("Updated PWM duty cycle on pin " << pin << " to " << dutyCycle);
74 return;
75 }
76
77 // Create new PWM channel
78 auto channel = std::make_unique<PWMChannel>(pin);
79 channel->dutyCycle = dutyCycle;
80 channel->frequency = frequency;
81
82 // Create Pin object for GPIO control
83 try {
84 channel->pinObj = std::make_unique<Pin>(pin, PinDirection::OUTPUT);
85 } catch (const std::exception& e) {
86 throw GpioAccessError("Failed to configure pin " + std::to_string(pin) +
87 " for PWM: " + e.what());
88 }
89
90 // Start PWM thread
91 channel->active = true;
92 PWMChannel* channelPtr = channel.get();
93 channel->pwmThread = std::thread(&PWMManager::pwmThreadFunction, channelPtr);
94
95 // Add to channels map
96 channels_[pin] = std::move(channel);
97
98 PIPINPP_LOG_INFO("Started PWM on pin " << pin << " (duty=" << dutyCycle
99 << ", freq=" << frequency << "Hz)");
100}

116bool PWMManager::stopPWM(int pin) {
117 std::lock_guard<std::mutex> lock(mutex_);
118
119 auto it = channels_.find(pin);
120 if (it == channels_.end()) {
121 return false; // PWM not active
122 }
123
124 // Remove channel (destructor will stop thread and clean up)
125 channels_.erase(it);
126
127 PIPINPP_LOG_INFO("Stopped PWM on pin " << pin);
128 return true;

159void PWMManager::pwmThreadFunction(PWMChannel* channel) {
160 PIPINPP_LOG_DEBUG("PWM thread started for pin " << channel->pin);
161
162 while (channel->active) {
163 int duty = channel->dutyCycle;
164 int freq = channel->frequency;
165
166 // Calculate period and timing
167 double periodUs = 1000000.0 / freq; // Period in microseconds
168 double onTimeUs = (periodUs * duty) / 255.0;
169
170 // Handle edge cases
171 if (duty == 0) {
172 // Always off
173 channel->pinObj->write(false);
174 std::this_thread::sleep_for(std::chrono::milliseconds(10));
175 continue;
176 } else if (duty == 255) {
177 // Always on
178 channel->pinObj->write(true);
179 std::this_thread::sleep_for(std::chrono::milliseconds(10));
180 continue;
181 }
182
183 // Generate PWM pulse
184 auto cycleStart = std::chrono::high_resolution_clock::now();
185
186 // HIGH period
187 channel->pinObj->write(true);
188 auto onTarget = cycleStart + std::chrono::nanoseconds(static_cast<long>(onTimeUs * 1000));
189 while (std::chrono::high_resolution_clock::now() < onTarget && channel->active) {
190 // Busy-wait for precision (necessary for software PWM)
191 std::this_thread::yield();
192 }
193
194 // LOW period
195 channel->pinObj->write(false);
196 auto offTarget = cycleStart + std::chrono::nanoseconds(static_cast<long>(periodUs * 1000));
197 while (std::chrono::high_resolution_clock::now() < offTarget && channel->active) {
198 // Busy-wait for precision
199 std::this_thread::yield();
200 }
201 }
202
203 // Ensure pin is LOW when stopping
204 if (channel->pinObj) {
205 channel->pinObj->write(false);
206 }
207
208 PIPINPP_LOG_DEBUG("PWM thread stopped for pin " << channel->pin);
209}
