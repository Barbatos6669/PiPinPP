33// Structure to store pin information
34struct PinInfo {
35 std::unique_ptr<Pin> pin;
36 ArduinoPinMode mode;
37 bool lastValue;
38};

40// Global storage for Arduino-style pins
41static std::unordered_map<int, PinInfo> globalPins;
42// Mutex to protect globalPins from concurrent access
43static std::mutex globalPinsMutex;

45void pinMode(int pin, int mode)
46{
47 std::lock_guard<std::mutex> lock(globalPinsMutex);
48 // Remove existing pin if it exists
49 globalPins.erase(pin);
50
51 // Create new pin with specified mode
52 // Exceptions will propagate naturally (InvalidPinError, GpioAccessError)
53 PinInfo pinInfo;
54 if (mode == OUTPUT) {
55 pinInfo.pin = std::make_unique<Pin>(pin, PinDirection::OUTPUT);
56 pinInfo.mode = ArduinoPinMode::OUTPUT;
57 pinInfo.lastValue = false;
58 } else if (mode == INPUT_PULLUP) {
59 pinInfo.pin = std::make_unique<Pin>(pin, PinMode::INPUT_PULLUP);
60 pinInfo.mode = ArduinoPinMode::INPUT_PULLUP;
61 pinInfo.lastValue = false;
62 } else if (mode == INPUT_PULLDOWN) {
63 pinInfo.pin = std::make_unique<Pin>(pin, PinMode::INPUT_PULLDOWN);
64 pinInfo.mode = ArduinoPinMode::INPUT_PULLDOWN;
65 pinInfo.lastValue = false;
66 } else {
67 pinInfo.pin = std::make_unique<Pin>(pin, PinDirection::INPUT);
68 pinInfo.mode = ArduinoPinMode::INPUT;
69 pinInfo.lastValue = false;
70 }
71
72 globalPins[pin] = std::move(pinInfo);
73
74 PIPINPP_LOG_INFO("pinMode: Set pin " << pin << " to "
75 << ((mode == OUTPUT) ? "OUTPUT" :
76 (mode == INPUT_PULLUP) ? "INPUT_PULLUP" :
77 (mode == INPUT_PULLDOWN) ? "INPUT_PULLDOWN" : "INPUT"));
78}

80void digitalWrite(int pin, bool value)
81{
82 std::lock_guard<std::mutex> lock(globalPinsMutex);
83 auto it = globalPins.find(pin);
84 if (it == globalPins.end()) {
85 throw InvalidPinError(pin, "Pin not initialized. Call pinMode() first.");
86 }
87
88 bool success = it->second.pin->write(value);
89 if (success) {
90 it->second.lastValue = value; // Track the last written value
91 } else {
92 throw GpioAccessError("pin " + std::to_string(pin), "Failed to write to GPIO pin");
93 }
94}

146long map(long x, long in_min, long in_max, long out_min, long out_max)
147{
148 // Prevent division by zero
149 if (in_max == in_min) {
150 return out_min; // Return minimum output if input range is zero
151 }
152 return (x - in_min) * (out_max - out_min) / (in_max - in_min) + out_min;
153}

195void attachInterrupt(int pin, void (*callback)(), int mode)
196{
197 // Convert Arduino interrupt mode to InterruptMode enum
198 InterruptMode intMode;
199 switch (mode) {
200 case RISING:
201 intMode = InterruptMode::RISING;
202 break;
203 case FALLING:
204 intMode = InterruptMode::FALLING;
205 break;
206 case CHANGE:
207 intMode = InterruptMode::CHANGE;
208 break;
209 default:
210 throw std::invalid_argument("Invalid interrupt mode: " + std::to_string(mode));
211 }
212
213 // Wrap the C-style function pointer in a std::function
214 InterruptCallback wrappedCallback = [callback]() {
215 if (callback) {
216 callback();
217 }
218 };
219
220 // Attach interrupt using InterruptManager
221 InterruptManager::getInstance().attachInterrupt(pin, wrappedCallback, intMode);
222
223 PIPINPP_LOG_INFO("attachInterrupt: Attached to pin " << pin
224 << " mode=" << mode);
225}

231}

288void digitalToggle(int pin)
289{
290 std::lock_guard<std::mutex> lock(globalPinsMutex);
291 auto it = globalPins.find(pin);
292 if (it == globalPins.end()) {
293 throw PinError("Pin " + std::to_string(pin) + " not initialized. Call pinMode() first.");
294 }
295
296 if (it->second.mode != ArduinoPinMode::OUTPUT) {
297 throw PinError("Pin " + std::to_string(pin) + " must be OUTPUT to toggle. "
298 "Current mode: " + std::to_string(static_cast<int>(it->second.mode)));
299 }
300
301 // Toggle efficiently - no need to read current value
302 bool newValue = !it->second.lastValue;
303 bool success = it->second.pin->write(newValue);
304 if (success) {
305 it->second.lastValue = newValue; // Only update if write succeeded
306 } else {
307 throw GpioAccessError("pin " + std::to_string(pin), "Failed to toggle GPIO pin");
308 }
309}
