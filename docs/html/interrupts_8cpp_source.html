81void InterruptManager::attachInterrupt(int pin, InterruptCallback callback,
82 InterruptMode mode, const std::string& chipname) {
83 // Validate pin number (0-27 for Raspberry Pi)
84 if (pin < 0 || pin > 27) {
85 throw InvalidPinError("Invalid pin number: " + std::to_string(pin) +
86 " (must be 0-27 for Raspberry Pi)");
87 }
88
89 if (!callback) {
90 throw std::invalid_argument("Interrupt callback cannot be null");
91 }
92
93 std::lock_guard<std::mutex> lock(mutex_);
94
95 // Check if interrupt already attached
96 if (handlers_.find(pin) != handlers_.end()) {
97 throw std::runtime_error("Interrupt already attached to pin " + std::to_string(pin));
98 }
99
100 // Create new handler
101 auto handler = std::make_unique<InterruptHandler>();
102 handler->pin = pin;
103 handler->callback = callback;
104 handler->mode = mode;
105
106 // Open GPIO chip
107 handler->chip = gpiod_chip_open(chipname.c_str());
108 if (!handler->chip) {
109 throw GpioAccessError("Failed to open GPIO chip: " + chipname);
110 }
111
112 // Configure line settings for edge detection
113 gpiod_line_settings* settings = gpiod_line_settings_new();
114 if (!settings) {
115 gpiod_chip_close(handler->chip);
116 throw std::runtime_error("Failed to create line settings");
117 }
118
119 // Set as input with edge detection
120 gpiod_line_settings_set_direction(settings, GPIOD_LINE_DIRECTION_INPUT);
121 gpiod_line_settings_set_edge_detection(settings, modeToEdge(mode));
122
123 // Create line config
124 gpiod_line_config* line_cfg = gpiod_line_config_new();
125 if (!line_cfg) {
126 gpiod_line_settings_free(settings);
127 gpiod_chip_close(handler->chip);
128 throw std::runtime_error("Failed to create line config");
129 }
130
131 unsigned int pin_offset = static_cast<unsigned int>(pin);
132 gpiod_line_config_add_line_settings(line_cfg, &pin_offset, 1, settings);
133
134 // Create request config
135 gpiod_request_config* req_cfg = gpiod_request_config_new();
136 if (!req_cfg) {
137 gpiod_line_config_free(line_cfg);
138 gpiod_line_settings_free(settings);
139 gpiod_chip_close(handler->chip);
140 throw std::runtime_error("Failed to create request config");
141 }
142
143 gpiod_request_config_set_consumer(req_cfg, "PiPinPP-Interrupt");
144 gpiod_request_config_set_event_buffer_size(req_cfg, 16);
145
146 // Request the line
147 handler->request = gpiod_chip_request_lines(handler->chip, req_cfg, line_cfg);
148
149 // Clean up config objects
150 gpiod_request_config_free(req_cfg);
151 gpiod_line_config_free(line_cfg);
152 gpiod_line_settings_free(settings);
153
154 if (!handler->request) {
155 gpiod_chip_close(handler->chip);
156 throw GpioAccessError("Failed to request line for pin " + std::to_string(pin));
157 }
158
159 // Create event buffer
160 handler->event_buffer = gpiod_edge_event_buffer_new(16);
161 if (!handler->event_buffer) {
162 gpiod_line_request_release(handler->request);
163 gpiod_chip_close(handler->chip);
164 throw std::runtime_error("Failed to create edge event buffer");
165 }
166
167 handler->active = true;
168
169 // Add to handlers map
170 handlers_[pin] = std::move(handler);
171
172 PIPINPP_LOG_INFO("Interrupt attached to pin " << pin);
173
174 // Start monitoring thread if not already running
175 if (!running_) {
176 startMonitoring();
177 } else {
178 // Wake up the monitor thread to reconfigure poll
179 char wake = '1';
180 ssize_t result = write(wakeup_pipe_[1], &wake, 1);
181 (void)result; // Intentionally ignore - wakeup is best-effort
182 }
183}

185bool InterruptManager::detachInterrupt(int pin) {
186 std::lock_guard<std::mutex> lock(mutex_);
187
188 auto it = handlers_.find(pin);
189 if (it == handlers_.end()) {
190 return false; // No interrupt attached
191 }
192
193 // Remove handler (destructor will clean up resources)
194 handlers_.erase(it);
195
196 PIPINPP_LOG_INFO("Interrupt detached from pin " << pin);
197
198 // Wake up the monitor thread to reconfigure poll
199 if (running_) {
200 char wake = '1';
201 ssize_t result = write(wakeup_pipe_[1], &wake, 1);
202 (void)result; // Intentionally ignore - wakeup is best-effort
203 }
204
205 return true;
206}

218void InterruptManager::startMonitoring() {
219 if (running_) {
220 return; // Already running
221 }
222
223 shutdown_requested_ = false;
224 running_ = true;
225 monitor_thread_ = std::thread(&InterruptManager::monitorThread, this);
226
227 PIPINPP_LOG_DEBUG("Interrupt monitoring thread started");
228}

230void InterruptManager::stopMonitoring() {
231 if (!running_) {
232 return; // Not running
233 }
234
235 PIPINPP_LOG_DEBUG("Stopping interrupt monitoring thread");
236
237 shutdown_requested_ = true;
238 running_ = false;
239
240 // Wake up the monitor thread
241 char wake = '1';
242 ssize_t result = write(wakeup_pipe_[1], &wake, 1);
243 (void)result; // Intentionally ignore - wakeup is best-effort
244
245 // Wait for thread to finish
246 if (monitor_thread_.joinable()) {
247 monitor_thread_.join();
248 }
249
250 PIPINPP_LOG_DEBUG("Interrupt monitoring thread stopped");
251}

253void InterruptManager::monitorThread() {
254 PIPINPP_LOG_DEBUG("Monitor thread running");
255
256 while (!shutdown_requested_) {
257 std::vector<pollfd> poll_fds;
258 std::vector<int> pin_map; // Maps poll_fds index to pin number
259
260 {
261 std::lock_guard<std::mutex> lock(mutex_);
262
263 // Add wakeup pipe
264 pollfd wakeup_fd;
265 wakeup_fd.fd = wakeup_pipe_[0];
266 wakeup_fd.events = POLLIN;
267 wakeup_fd.revents = 0;
268 poll_fds.push_back(wakeup_fd);
269 pin_map.push_back(-1); // -1 indicates wakeup pipe
270
271 // Add all active interrupt handlers
272 for (auto& pair : handlers_) {
273 if (pair.second->active) {
274 pollfd pfd;
275 pfd.fd = gpiod_line_request_get_fd(pair.second->request);
276 pfd.events = POLLIN;
277 pfd.revents = 0;
278 poll_fds.push_back(pfd);
279 pin_map.push_back(pair.first);
280 }
281 }
282 }
283
284 if (poll_fds.size() == 1) {
285 // Only wakeup pipe, no active interrupts
286 // Wait for new interrupts to be attached
287 poll(poll_fds.data(), poll_fds.size(), -1);
288
289 // Clear wakeup pipe
290 char buf[16];
291 ssize_t bytes_read = read(wakeup_pipe_[0], buf, sizeof(buf));
292 (void)bytes_read; // Intentionally ignore - just clearing the pipe
293 continue;
294 }
295
296 // Wait for events (timeout 100ms to check shutdown flag periodically)
297 int ret = poll(poll_fds.data(), poll_fds.size(), 100);
298
299 if (ret < 0) {
300 if (errno == EINTR) {
301 continue; // Interrupted by signal, retry
302 }
303 PIPINPP_LOG_ERROR("poll() failed: " << strerror(errno));
304 break;
305 }
306
307 if (ret == 0) {
308 continue; // Timeout, check shutdown flag
309 }
310
311 // Check wakeup pipe first
312 if (poll_fds[0].revents & POLLIN) {
313 char buf[16];
314 ssize_t bytes_read = read(wakeup_pipe_[0], buf, sizeof(buf));
315 (void)bytes_read; // Intentionally ignore - just clearing the pipe
316 // Reconfigure poll on next iteration
317 continue;
318 }
319
320 // Process events on interrupt pins
321 for (size_t i = 1; i < poll_fds.size(); ++i) {
322 if (poll_fds[i].revents & POLLIN) {
323 int pin = pin_map[i];
324
325 std::unique_lock<std::mutex> lock(mutex_);
326 auto it = handlers_.find(pin);
327 if (it == handlers_.end() || !it->second->active) {
328 continue; // Handler was removed
329 }
330
331 auto& handler = it->second;
332 lock.unlock();
333
334 // Read edge events
335 int num_events = gpiod_line_request_read_edge_events(
336 handler->request, handler->event_buffer,
337 gpiod_edge_event_buffer_get_capacity(handler->event_buffer)
338 );
339
340 if (num_events > 0) {
341 // Process all events
342 for (int j = 0; j < num_events; ++j) {
343 gpiod_edge_event* event = gpiod_edge_event_buffer_get_event(
344 handler->event_buffer, j
345 );
346
347 if (event) {
348 // Invoke callback
349 try {
350 handler->callback();
351 } catch (const std::exception& e) {
352 PIPINPP_LOG_ERROR("Exception in interrupt callback for pin "
353 << pin << ": " << e.what());
354 } catch (...) {
355 PIPINPP_LOG_ERROR("Unknown exception in interrupt callback for pin " << pin);
356 }
357 }
358 }
359 }
360 }
361 }
362 }
363
364 PIPINPP_LOG_DEBUG("Monitor thread exiting");
365}

367gpiod_line_edge InterruptManager::modeToEdge(InterruptMode mode) {
368 switch (mode) {
369 case InterruptMode::RISING:
370 return GPIOD_LINE_EDGE_RISING;
371 case InterruptMode::FALLING:
372 return GPIOD_LINE_EDGE_FALLING;
373 case InterruptMode::CHANGE:
374 return GPIOD_LINE_EDGE_BOTH;
375 default:
376 return GPIOD_LINE_EDGE_BOTH;
377 }
